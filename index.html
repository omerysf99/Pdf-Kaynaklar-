// ANA TIKLAMA FONKSİYONU (Hem Mouse Hem Touch Destekli) function handleTap(e) { // Sayfa kaymasını ve varsayılan olayları engelle if (e.type === 'touchstart') e.preventDefault(); if (energy > 0) { // Değerleri Güncelle coins += 1; energy -= 1; // Ekranı Güncelle (Coin miktarını anlık yazdır) balanceDisplay.innerText = coins.toLocaleString(); energyNumDisplay.innerText = energy; energyBarDisplay.style.width = (energy / maxEnergy * 100) + "%"; // Seviye Barını Güncelle levelBar.style.width = Math.min(100, (coins / 50)) + "%"; // Telegram Titreşimi if (window.Telegram && tg.HapticFeedback) { tg.HapticFeedback.impactOccurred('medium'); } // Görsel Efekt (+1 fırlatması) const clientX = e.type === 'touchstart' ? e.touches[0].clientX : e.clientX; const clientY = e.type === 'touchstart' ? e.touches[0].clientY : e.clientY; createPlusOne(clientX, clientY); } else { // Enerji bittiyse kullanıcıya uyarı ver (isteğe bağlı) console.log("Enerji bitti!"); } } // Hem dokunma hem tıklama için dinleyici ekle spyderEmoji.addEventListener('touchstart', handleTap); spyderEmoji.addEventListener('mousedown', handleTap);
